\pgfmathsetmacro{\scale}{1.0}
\begin{tikzpicture}[thick, scale=\scale, hexagonXYZ]
	
	\pgfmathtruncatemacro{\width}{4}
	\pgfmathtruncatemacro{\height}{4}
	
	\foreach \x in {1,...,\width}{
		\foreach \y in {1,...,\height}{
			\node [ draw
			      , hexagon
			      , inner sep=0
			      , minimum width=\scale * 1cm
			      , help lines
			      ]
			      at (\x, \y, 0)
			      {};
		}
	}
	
	\begin{scope}[inner sep=0]
		\node [hexagon] at (2, 2, 0) (n1) {};
		\node [hexagon] at (3, 2, 0) (n2) {};
		\node [hexagon] at (3, 3, 0) (n3) {};
	\end{scope}
	
	\begin{scope} [thick]
		\draw [->]
		      (n1.side south)
		      -- node [anchor=north, inner sep=0.2cm] {\contour{white}{X}}
		      (n2.side south west);
		\draw [->]
		      (n2.side north east)
		      -- node [anchor=west, inner sep=0.2cm] {\contour{white}{Y}}
		      (n3.side east);
		\draw [->]
		      (n3.side west)
		      -- node [anchor=south east, inner sep=0.1cm] {\contour{white}{Z}}
		      (n1.side north);
	\end{scope}
	
\end{tikzpicture}
